Can include **markdown**. {{< /faq >}} Parameters: - question (required): The FAQ question text The FAQ schema JSON-LD is automatically generated at the end of the page by the baseof template when FAQs are present. */ -}} {{- $question := .Get "question" -}} {{- $answer := .Inner | .Page.RenderString -}} {{- if not $question -}} {{- errorf "FAQ shortcode requires a 'question' parameter. File: %s" .Page.File.Path -}} {{- end -}} {{- /* Generate a unique ID for this FAQ item */ -}} {{- $faqId := printf "faq-%s" ($question | anchorize) -}} {{- /* Add this FAQ to page scratch for schema generation */ -}} {{- $faqData := dict "question" $question "answer" ($answer | plainify) -}} {{- .Page.Scratch.Add "faqs" (slice $faqData) -}}
